# Notify

## Setup

```js
import { Notify } from '@dear-ui/all/core/notification'

Notify({
  config: {
    text: '通知内容',
    id: 'ID',
    title: '通知标题',
    type: 'white', //[success error normal warn black white] default 'desc'
    timer: -1,
    onClickTip: (e) => {console.log(e)},
    actionText: '有 action 时显示的文字',
  }
})
```

## 通过 EventEmitter 广播

```js
import { EventEmitter } from '@mini-code/base-func'
import { NotifyEvent } from '@dear-ui/all/core/notification'

EventEmitter.emit(NotifyEvent, {
  text: '通过 EventEmitter 广播通知',
  title: '通知标题',
  type: 'success', //[success error normal warn black white] default 'desc'
  lifecycle: 7,
  onClickTip: (e) => {console.log(e)},
  actionText: '有 action 时显示的文字',
})
```

## 取消广播

```js
import { Notify, CancelNotify } from '@dear-ui/all'

const NotifyID = Notify({

})

CancelNotify(NotifyID);
```
