# TreeList

## Basic Usage

```jsx live noInline
import { TreeList } from '@deer-ui/enhance-ui';

function Example() {
  const treeData = [
    {
      title: '树形1',
      value: 'value1',
      id: 'value1',
      child: [{
        title: '树形1-1',
        value: 'value1-1',
        id: 'value1-1',
        child: [{
          value: 'value1-1-1',
          id: 'value1-1-1',
          title: '树形1-1-1',
        }]
      }, {
        title: '树形1-2',
        value: 'value1-2',
        id: 'value1-2',
        child: [{
          id: 'value1-2-1',
          value: 'value1-2-1',
          title: '树形1-2-1',
        }, {
          title: '树形1-2-2',
          id: 'value1-2-2',
          value: 'value1-2-2',
          child: [{
            id: 'value1-2-2-1',
            value: 'value1-2-2-1',
            title: '树形1-2-2-1',
          }]
        }]
      }],
    },
    {
      title: '树形2',
      id: 'value2',
      value: 'value2',
      child: [{
        title: '树形2-1',
        id: 'value2-1',
        value: 'value2-1',
        child: [{
          title: '树形2-1-1',
          id: 'value2-1-1',
          value: 'value2-1-1',
        }]
      }, {
        title: '树形2-2',
        id: 'value2-2',
        value: 'value2-2',
        child: [{
          title: '树形2-2-1',
          id: 'value2-2-1',
          value: 'value2-2-1',
        }]
      }],
    },
  ]
  return (
    <TreeList
      onChange={val => {
        console.log(val)
      }}
      defaultValue={{
        'value2-2': 'value2-2'
      }}
      treeData={treeData} />
  )
}
render(<Example />);
```

## 调节 TreeListData 的字段与组件内部字段的对应关系

```js
<TreeList
  fieldMapper={{
    child: 'Children',
    id: 'ID',
    value: 'Value',
    title: 'Name',
  }}
  onChange={val => {
    console.log(val)
  }}
  treeData={treeData} />
```
