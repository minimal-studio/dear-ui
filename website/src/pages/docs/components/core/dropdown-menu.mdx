import { useState } from 'react'
import { Playground, Props } from 'docz'
import { Dropdown } from '@deer-ui/core/dropdown';

# Dropdown

> 基于 selector 的下拉选择器组件，支持多选

## Basic Usage

> value 由父组件控制

<Playground>
  {
    () => {
      const values = {
        value1: 'Item 1',
        value2: 'Item 2',
        value3: 'Item 3',
        value4: 'Item 1',
        value22: 'Item 2',
        value31: 'Item 3',
        value12: 'Item 1',
        value241: 'Item 2',
        value312: 'Item 3',
        value151: 'Item 1',
        value112: 'Item 2',
        value1233: 'Item 3',
        value51: 'Item 1',
        value512: 'Item 2',
        value37: 'Item 3',
        value1754: 'Item 1',
        value245: 'Item 2',
        value4573: 'Item 3',
      }
      const [value, setValue] = useState('value1')
      return (
        <div className="mu10">
          <Dropdown
            onChange={val => {
              setValue(val)
            }}
            value={value}
            values={values}
          />
          <span className="ml10">输出值 {value}</span>
        </div>
      )
    }
  }
</Playground>

## Multiple 多选

<Playground>
  {
    () => {
      const values = {

## 通过 ref 改变组件内部的值

<Playground>
  {
    () => {
      const values = {
        value1: 'Item 1',
        value2: 'Item 2',
        value3: 'Item 3',
      }
      const [value, setValue] = useState('value1')
      let dropdownRef;
      return (
        <div className="mu10">
          <span className="btn theme mb10"
            onClick={e => {
              dropdownRef.changeValue('value3')
            }}>把 Dropdown 实例中的更改为 value3</span>
          <hr />
          <Dropdown
            ref={e => dropdownRef = e}
            onChange={val => {
              setValue(val)
            }}
            value={value}
            values={values}
          />
          <span className="ml10">输出值 {value}</span>
        </div>
      )
    }
  }
</Playground>

## 输出 Number 类型的值

<Playground>
  {
    () => {
      const valuesNumber = {
        0: '数字0',
        1: '数字1',
        2: '数字2',
        3: '数字3',
      }
      const [value, setValue] = useState(0)
      const [valueMul, setValueMul] = useState([])
      return (
        <div className="mu10">
          <Dropdown
            onChange={val => {
              setValue(val)
            }}
            isNum
            defaultValue={value}
            values={valuesNumber}
          />
          <span className="ml10">输出值 {value}, 类型 {typeof value}</span>
          <hr />
          <h4>多选</h4>
          <Dropdown
            onChange={val => {
              setValueMul(val)
            }}
            isMultiple
            values={valuesNumber}
          />
          <span className="ml10">输出值 {valueMul.join(' ')}</span>
        </div>
      )
    }
  }
</Playground>

## 把 Dropdown 渲染在 react root 之外

### 默认监听 document.body 的 onScroll 事件，实现滚动隐藏 Dropdown

<Playground>

          <Dropdown
            outside
            values={values}
          />
        </div>
      )
    }
  }
</Playground>

### 监听指定元素的 onScroll 事件

监听 Dropdown 的父级元素的 onScroll 事件并隐藏 Dropdown

<Playground>
  {
    () => {
      const values = {
        value1: 'Item 1',
        value2: 'Item 2',
        value3: 'Item 3',
      }
      let containerRef;
      return (
        <div className="mu10" 
          style={{
            height: 100,
            overflow: 'auto'
          }}
          ref={e => containerRef = e}>
          <div style={{
            height: 200
          }}>
            <Dropdown
              outside
              scrollElem={() => containerRef}
              values={values}
            />
          </div>
        </div>
      )
    }
  }
</Playground>
