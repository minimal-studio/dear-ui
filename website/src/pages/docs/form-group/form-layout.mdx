# FormLayout

```jsx live noInline
import React, { useState } from 'react';
import { FormLayout, Button } from '@deer-ui/core';
import formOptions from 'formOptions';

const FormLayoutDemo = () => {
  const [layout, setLayout] = useState('horizontal');
  const [resDesc, setResDesc] = useState('');

  const formBtns = [
    {
      action: (formRef, actingRef) => {
        console.log(formRef.value, actingRef);
        this.submit(formRef.value, actingRef);
      },
      type: 'submit',
      text: '测试按钮',
      actingRef: 'forTest'
    },
  ];

  const submit = (formData, actingRef) => {
    // 模拟获取数据
    setTimeout(() => {
      this.setState({
        hasErr: false,
        resDesc: '成功'
      });
    }, 800);
  }

  const isHorizontal = layout === 'horizontal';
  console.log(layout)
  return (
    <>
      <h3>默认水平布局，可以切换成垂直布局（适配移动端）</h3>
      <Button
        onClick={e => {
          setLayout(isHorizontal ? 'vertical' : 'horizontal')
        }}>
        切换成{isHorizontal ? '垂直' : '水平'}布局
      </Button>
      <FormLayout
        layout={layout}
        formBtns={formBtns}
        formOptions={formOptions}/>
    </>
  );
}
render(<FormLayoutDemo />);
```

## 动态表单

> 更改其中某个配置的选择值

```jsx live noInline
import { useState } from 'react'
import { FormLayout, Button, Loading } from '@deer-ui/core';
import formOptions from 'formOptions';

function Example() {
  const valuesMapper = {
    value1: {
      forValue11: 'value1-1 的动态值',
      forValue12: 'value1-2 的动态值',
      forValue13: 'value1-3 的动态值',
    },
    value2: {
      forValue21: 'value2-1 的动态值',
      forValue22: 'value2-2 的动态值',
      forValue23: 'value2-3 的动态值',
    },
    value3: {
      forValue31: 'value3-1 的动态值',
      forValue32: 'value3-2 的动态值',
      forValue33: 'value3-3 的动态值',
    },
  }
  const formOptions1 = [
    {
      ref: 'ref1',
      type: 'radio',
      title: '单选控件',
      values: {
        value1: 'value1',
        value2: 'value2',
        value3: 'value3',
      }
    },
    {
      ref: 'ref2',
      type: 'radio',
      title: '选择控件',
      values: valuesMapper['value1']
    },
    {
      ref: 'ref3',
      type: 'select',
      title: '选择控件',
      isMultiple: true,
      values: valuesMapper['value1']
    },
  ]
  const [state, setState] = useState({
    formOptions: [...formOptions1],
    loading: false
  })
  let formRef = {};
  return (
    <div>
      <h4>动态表单</h4>
      <span className="btn theme mr10" onClick={e => setState({
        ...state,
        loading: !state.loading
      })}>模拟异步获取表单条件</span>
      <span className="btn red" onClick={e => {
        formRef.showResDesc({
          hasErr: false,
          resDesc: '成功'
        })
      }}>使用 FormLayout 的 showResDesc 接口显示 Toast 提示</span>
      <div>
        <Loading loading={state.loading}>
          {
            !state.loading && (
              <FormLayout
                formBtns={[
                  {
                    action: (formRef, actingRef) => {
                      console.log(formRef.value, actingRef);
                    },
                    type: 'submit',
                    text: '测试按钮, 带预检查',
                    actingRef: 'forTest'
                  },
                  {
                    action: () => {
                      formRef.clearValue();
                    },
                    text: '清空内容'
                  }
                ]}
                ref={e => formRef = e}
                onChange={(value, ref, val) => {
                  if(ref == 'ref1') {
                    const nextState = [...state.formOptions];
                    // 更改新 formOptions 的目标的 values
                    nextState[1].values = {...valuesMapper[val]};
                    nextState[2].values = valuesMapper[val];
                    setState({
                      ...state,
                      formOptions: nextState
                    })
                  }
                }}
                formOptions={state.formOptions}/>
            )
          }
        </Loading>
      </div>
    </div>
  )
}
render(<Example />);
```

## formOptions 后赋值

```jsx live noInline
import { useState } from 'react'
import { FormLayout, Button, Loading } from '@deer-ui/core';
import formOptions from 'formOptions';

function Example() {
  const valuesMapper = {
    value1: {
      forValue11: 'value1-1 的动态值',
      forValue12: 'value1-2 的动态值',
      forValue13: 'value1-3 的动态值',
    },
    value2: {
      forValue21: 'value2-1 的动态值',
      forValue22: 'value2-2 的动态值',
      forValue23: 'value2-3 的动态值',
    },
    value3: {
      forValue31: 'value3-1 的动态值',
      forValue32: 'value3-2 的动态值',
      forValue33: 'value3-3 的动态值',
    },
  }
  const formOptions1 = [
    {
      ref: 'ref1',
      type: 'radio',
      title: '单选控件',
      defaultValue: 'value1',
      values: {
        value1: 'value1',
        value2: 'value2',
        value3: 'value3',
      }
    },
    {
      ref: 'ref2',
      type: 'radio',
      title: '选择控件',
      values: valuesMapper['value1']
    },
    {
      ref: 'ref3',
      type: 'select',
      title: '选择控件',
      isMultiple: true,
      values: valuesMapper['value1']
    },
  ]
  let formRef = {};
  const [formOptions, setFormOptions] = useState([]);
  return (
    <div>
```
