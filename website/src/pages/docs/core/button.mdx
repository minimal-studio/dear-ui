# Button

## Status Button

根据不同的 status 决定按钮颜色

```jsx live noInline
import { Button } from '@deer-ui/core'

function Example() {
  return (
    <>
      <Button
        status="primary"
        className="mr10 mb10">
        Primary
      </Button>
      <Button
        color="default"
        className="mr10 mb10">
        Default
      </Button>
      <Button
        color="danger"
        className="mr10 mb10">
        Danger
      </Button>
      <Button
        disabled
        status="warn"
        className="mr10 mb10">
        Disabled
      </Button>
      <Button
        loading
        color="green"
        className="mr10 mb10">
        Loading
      </Button>
      <Button
        status="link"
        className="mr10 mb10">
        Link
      </Button>
    </>
  )
}
render(<Example />);
```

## Normal Button

```jsx live noInline
import { Button } from '@deer-ui/core'
import { ExampleColors, ForEachColor } from '@deer-ui/core/utils';

function Example() {
  return (
    <div>
      {
        ForEachColor(ExampleColors)((color) => (
          <Button key={color}
            color={color}
            className="mr10 mb10">
            {color.toUpperCase()}
          </Button>
        ))
      }
    </div>
  )
}
render(<Example />);
```

## With Icon

```jsx live noInline
import { Button } from '@deer-ui/core'
import { ExampleColors, ForEachColor } from '@deer-ui/core/utils';

function Example() {
  const icons = [
    {icon: 'envelope-open', s: 'r'},
    {icon: 'paper-plane', s: 'r'},
    {icon: 'align-right'},
    {icon: 'bell', s: 'r'},
    {icon: 'check-circle', s: 'r'},
  ];
  return (
    <div>
      {
        ForEachColor(ExampleColors)((color, idx) => {
          return (
            <Button key={color}
              color={color}
              hola
              icon={icons[idx].icon}
              s={icons[idx].s}
              className="mr10 mb10">
              {color.toUpperCase()}
            </Button>
          )
        })
      }
    </div>
  )
}
render(<Example />);
```
