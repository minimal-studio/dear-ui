# Radio

> 用于多个选项中的单选操作

## Basic Usage

```jsx live noInline
import { useState } from 'react';
import { Radio } from '@deer-ui/core';

function Example() {
  const values1 = {
    value1: '显示的值1',
    value2: '显示的值2',
    value3: '显示的值3',
  }
  const [value, setValue] = useState('value1')
  return (
    <div>
      <Radio
        onChange={val => {
          console.log(val)
          setValue(val)
        }}
        value={value}
        values={values1}
      />
      <hr />
      <span className="mr10">输出值： {value}</span>
    </div>
  )
}
render(<Example />);
```

## 通过 ref 改变值

> 也可以同时设置每个 item 的 style

```jsx live noInline
import { useState } from 'react';
import { Radio, Button } from '@deer-ui/core';

function Example() {
  let radio1;
  const values1 = {
    value1: '显示的值1',
    value2: '显示的值2',
    value3: '显示的值3',
  }
  const [value, setValue] = useState('value1')
  return (
    <div>
      <Button className="btn theme" onClick={e => radio1.changeValue('value3')}>点击改变为 value3</Button>
      <hr/>
      <Radio
        ref={e => radio1 = e}
        onChange={val => {
          console.log(val)
          setValue(val)
        }}
        value={value}
        values={values1}
      />
    </div>
  )
}
render(<Example />);
```

## 任意显示内容

```jsx live noInline
import { useState } from 'react';
import { Radio } from '@deer-ui/core';

function Example() {
  const values2 = {
    value1: (
      <h1>标题</h1>
    ),
    value2: (
      <div>
        <img src="https://m.360buyimg.com/pop/jfs/t22765/41/1722519893/139051/da2d7354/5b6948b5Nb66bd80d.png"/>
        <span>下标题</span>
      </div>
    ),
    value3: '显示的值3',
  }
  const [value, setValue] = useState('value1')
  return (
    <div>
      <Radio
        onChange={val => {
          setValue(val)
        }}
        value={value}
        itemStyle={{
          border: '1px solid #EEE',
          maxWidth: 100
        }}
        values={values2}
      />
    </div>
  )
}
render(<Example />);
```
