# SwitchBtn

切换按钮

## Basic Usage

```jsx live noInline
import { useState } from 'react';
import { SwitchBtn } from '@deer-ui/core';

function Example() {
  const [activeIdx, setIdx] = useState('btn1')
  return (
    <SwitchBtn
      onSwitch={val => setIdx(val)}
      activeIdx={activeIdx}
      btns={{
        btn1: '选项1',
        btn2: '选项2',
        btn3: '选项3',
      }} />
  )
}
render(<Example />);
```

## 输出 Number 类型的值

```jsx live noInline
import { useState } from 'react';
import { SwitchBtn } from '@deer-ui/core';

function Example() {
  const [activeIdx, setIdx] = useState(0)
  return (
    <SwitchBtn
      onSwitch={val => setIdx(val)}
      activeIdx={activeIdx}
      isNum
      btns={{
        0: '选项1',
        1: '选项2',
        2: '选项3',
      }} />
  )
}
render(<Example />);
```

## 禁用状态

```jsx live noInline
import { useState } from 'react';
import { SwitchBtn } from '@deer-ui/core';

function Example() {
  const [activeIdx, setIdx] = useState('btn1')
  console.log(activeIdx)
  return (
    <SwitchBtn
      onSwitch={val => setIdx(val)}
      disabled
      activeIdx={'btn1'}
      btns={{
        btn1: '选项1',
        btn2: '选项2',
        btn3: '选项3',
      }} />
  )
}
render(<Example />);
```
