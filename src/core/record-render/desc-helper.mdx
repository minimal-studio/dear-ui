import { Playground, Props } from 'docz'
import { State, Toggle } from 'react-powerplug'
import { DescHelper } from './';

# DescHelper

> 用于渲染列表中详细数据的组件

## Basic Usage

<Playground>
  {
    () => {
      let longText = '';
      for(let i = 100; i >= 0; i--) {
        longText = longText + 'Long ';
      }

      const keyMapper = [
        {
          key: 'username',
          namesMapper: {
            alex: '埃里克斯',
            chili: '吃梨',
            dove: '德芙',
          }
        },
        {key: 'age'},
        {key: 'add'},
        // block 为占用一行
        {key: 'longText'},
        {key: 'desc2'},
        {key: 'desc3'},
        {key: 'desc4'},
      ]

      const records = [
        {
          username: 'alex',
          age: '100',
          add: 'cn',
          longText: longText,
        },
        {
          username: 'chili',
          age: '102',
          add: 'cn',
        },
        {
          username: 'dove',
          age: '50',
          add: 'cn',
        },
        {
          username: 'susam',
          age: '20',
          add: 'uk',
        },
      ]
      return (
        <State initial={{ activeRecordIdx: 0 }}>
          {
            ({ state, setState }) => (
              <div>
                {
                  records.map((item, idx) => {
                    return (
                      <span
                        className="btn theme flat m5"
                        onClick={e => {
                          setState({
                            activeRecordIdx: idx
                          })
                        }}
                        key={idx}>点击查看{item.username}的详情</span>
                    )
                  })
                }
                <DescHelper
                  keyMapper={keyMapper}
                  record={records[state.activeRecordIdx]} />
              </div>
            )
          }
        </State>
      )
    }
  }
</Playground>
