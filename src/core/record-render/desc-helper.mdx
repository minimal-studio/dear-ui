## 在父组件定义渲染配置

```js
class ForTable extends React.Component {
  constructor(props) {
    super(props);

    let longText = '';
    for(let i = 100; i >= 0; i--) {
      longText = longText + 'Long ';
    }

    this.keyMapper = [
      {
        key: 'username',
        namesMapper: {
          alex: '埃里克斯',
          chili: '吃梨',
          dove: '德芙',
        }
      },
      {key: 'age'},
      {key: 'add'},
      // block 为占用一行
      {key: 'longText'},
      {key: 'desc2'},
      {key: 'desc3'},
      {key: 'desc4'},
    ]

    this.records = [
      {
        username: 'alex',
        age: '100',
        add: 'cn',
        longText: longText,
      },
      {
        username: 'chili',
        age: '102',
        add: 'cn',
      },
      {
        username: 'dove',
        age: '50',
        add: 'cn',
      },
      {
        username: 'susam',
        age: '20',
        add: 'uk',
      },
    ]

    this.state = {
      activeRecordIdx: 0
    }
  }
  render() {
    const { activeRecordIdx } = this.state;
    return (
      <div>
        {
          this.records.map((item, idx) => {
            return (
              <span
                className="btn theme flat m5"
                onClick={e => {
                  this.setState({
                    activeRecordIdx: idx
                  })
                }}
                key={idx}>点击查看{item.username}的详情</span>
            )
          })
        }
        <DescHelper
          keyMapper={this.keyMapper}
          record={this.records[activeRecordIdx]} />
      </div>
    )
  }
}

<ForTable />
```
