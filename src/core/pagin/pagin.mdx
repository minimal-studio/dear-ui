import { useState } from 'react'
import { Playground, Props } from 'docz'
import { Pagination } from './';

# Pagination

## Basic Usage

> 输入自定义 key 的数据，输出对应的数据结构

<Playground>
  {
    () => {
      /** 支持自定义数据结构 */
      const [pagingInfo, setPagin] = useState({
        Active: true,
        Total: 1000,
        PIdx: 0,
        PSize: 10
      });
      const paginTransfer = {
        active: 'Active',
        total: 'Total',
        pIdx: 'PIdx',
        pSize: 'PSize'
      }
      return (
        <Pagination
          onPagin={nextPagin => {
            console.log(nextPagin)
            setPagin(nextPagin)
          }}
          infoMapper={paginTransfer}
          pagingInfo={pagingInfo}/>
      )
    }
  }
</Playground>
