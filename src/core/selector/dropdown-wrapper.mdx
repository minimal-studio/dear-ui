import { Playground, Props } from 'docz'
import DropdownWrapper from './dropdown-wrapper';

# DropdownWrapper

> 通用的下拉组件，可以渲染任意子内容

## Basic Usage

<Playground>
  {
    () => {
      return (
        <>
          <DropdownWrapper menuTitle="标题" overlay={e => {
            return (
              <div className="p10" style={{width: 400}}>
                任意内容
              </div>
            )
          }} />
          <span className="ms10"></span>
          <DropdownWrapper menuTitle="鼠标移入" trigger="hover" overlay={e => {
            return (
              <div className="p10" style={{width: 400}}>
                任意内容
              </div>
            )
          }} />
        </>
      )
    }
  }
</Playground>

## 自定义外层显示内容

<Playground>
  {
    () => {
      return (
        <DropdownWrapper overlay={() => {
          return (
            <div className="p10" style={{width: 400}}>
              弹出内容
            </div>
          )
        }}>
          {
            (options) => {
              return (
                <div>显示内容</div>
              )
            }
          }
        </DropdownWrapper>
      )
    }
  }
</Playground>

## 渲染在 root 以外

<Playground>
  {
    () => {
      return (
        <div>
          <DropdownWrapper outside overlay={(helper) => {
            return (
              <div className="p10" style={{width: 400}}>
                弹出内容, <span className="link" onClick={e => {
                  helper.hide()
                }}>点击关闭</span>
              </div>
            )
          }}>
            {
              (options) => {
                return (
                  <div>显示内容</div>
                )
              }
            }
          </DropdownWrapper>
          <span className="ms10"></span>
          <DropdownWrapper outside overlay={(helper) => {
            return (

                  helper.hide()
                }}>点击关闭</span>
              </div>
            )
          }}>
            {
              (options) => {
                return (
                  <div>显示内容</div>
                )
              }
            }
          </DropdownWrapper>
        </div>
      )
    }
  }
</Playground>
