# Usage

## 类别

- src/core  核心组件集
- src/other 其他组件集合库

## /core 按需加载

```jsx static

import Multiple from 'dear-ui/core/multiple-selector';
import Avatar from 'dear-ui/core/avatar';
import Captcha from 'dear-ui/core/captcha';
import ChartCom from 'dear-ui/core/chart-dom';
import Carousel from 'dear-ui/core/carousel/carousel';
import Loading from 'dear-ui/core/loading';
import SwitchBtn from 'dear-ui/core/switch-button';
import TipPanel from 'dear-ui/core/tip-panel';
import Toast from 'dear-ui/core/toast';
import Ranger from 'dear-ui/core/range-selector';
import Pagination from 'dear-ui/core/paging-button';
import TipBtn from 'dear-ui/core/tip-button';
import Icon from 'dear-ui/core/icon';
import Label from 'dear-ui/core/label';

import { Notify, CancelNotify, Notification } from 'dear-ui/core/notification';

import { Popover, Pop, GlobalPopover, PopoverEntity } from 'dear-ui/core/popover';
```
